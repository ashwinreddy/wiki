$$
\mathbb{V}[X] \triangleq \mathbb{E}[(X - \mu)^2]
$$

\begin{align\*}
\mathbb{V}[X] &= \mathbb{E}[(X - \mu)^2] \\\\\
&= \mathbb{E}[X^2 - 2\mu X + \mu^2] \\\\\
&= \mathbb{E}[X^2] - 2\mu\mathbb{E}[X] + \mu^2 \\\\\
&= \mathbb{E}[X^2] - \mu^2
\end{align\*}

\begin{equation}
\mathbb{V}[X] = \mathbb{E}[X^2] - \mathbb{E}[X]^2
\end{equation}

$$
\mathbb{V}[X+Y]= \mathbb{V}[X] + \mathbb{V}[Y]
$$
