\begin{equation}
{\displaystyle \mathbf {B} =\nabla \times \mathbf {A} \,,\quad \mathbf {E} =-\nabla \phi -{\frac {\partial \mathbf {A} }{\partial t}}\,,}

\begin{equation}
{\displaystyle {\begin{aligned}\nabla ^{2}\phi -{\frac {1}{c^{2}}}{\frac {\partial ^{2}\phi }{\partial t^{2}}}&=-{\frac {\rho }{\epsilon _{0}}}\\\\\nabla ^{2}\vec {A} -{\frac {1}{c^{2}}}{\frac {\partial ^{2}\vec {A} }{\partial t^{2}}}&=-\mu _{0}\vec {j} \end{aligned}}}
\end{equation}

\begin{align*}
